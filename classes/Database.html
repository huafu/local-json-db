Example:

  var db = new Database();
  var record = db.createRecord({name: 'Huafu'});
  record.age = 31;
  db.updateRecord(record);
  console.log(db.find('user', record.id));
  // will output {id: 1, name: 'Huafu', age: 31}
  db.save();
